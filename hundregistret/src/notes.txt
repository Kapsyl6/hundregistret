import java.util.Comparator;

public class NumberComparator implements Comparator <Number> {

    public int compare(Number first, Number second) {
        if(first.getValue() < second.getValue())
            return -1;
        if(first.getValue() < second.getValue())
                return 1;
            
        return 0;

    }

    
}
